local activeTags = {}

local function addObjectTag(obj, text, color)
    if not obj or not obj.Parent then return end
    for i, tag in ipairs(activeTags) do
        if tag.Adornee == obj then
            tag:Destroy()
            table.remove(activeTags, i)
            break
        end
    end
    local bill = Instance.new("BillboardGui")
    bill.Name = "STN_ObjectTag"
    bill.AlwaysOnTop = true
    bill.Size = UDim2.new(0, 120, 0, 30)
    bill.StudsOffset = Vector3.new(0, 2, 0)
    bill.Parent = obj
    local label = Instance.new("TextLabel", bill)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = color
    label.TextSize = 14
    label.Font = Enum.Font.GothamBold
    label.TextStrokeTransparency = 0.3
    label.TextStrokeColor3 = Color3.new(0, 0, 0)
    table.insert(activeTags, bill)
end

local function removeTagsByCondition(condition)
    for i = #activeTags, 1, -1 do
        local tag = activeTags[i]
        if condition(tag) then
            tag:Destroy()
            table.remove(activeTags, i)
        end
    end
end

local function removeAllTags()
    for _, tag in pairs(activeTags) do
        tag:Destroy()
    end
    activeTags = {}
end

local instantInteractActive = false
local instantInteractConnection = nil
local instantInteractTrackConnection = nil
local originalHoldDurations = {}

local function saveOriginalHoldDuration(prompt)
    if not originalHoldDurations[prompt] then
        originalHoldDurations[prompt] = prompt.HoldDuration
    end
end

local function restoreAllPrompts()
    for prompt, orig in pairs(originalHoldDurations) do
        if prompt and prompt.Parent then
            prompt.HoldDuration = orig
        end
    end
end

local function onDescendantAdded(descendant)
    if instantInteractActive and descendant:IsA("ProximityPrompt") then
        saveOriginalHoldDuration(descendant)
        descendant.HoldDuration = 0
    end
end

local function enableInstantInteract()
    if instantInteractActive then return end
    instantInteractActive = true
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("ProximityPrompt") then
            saveOriginalHoldDuration(v)
            v.HoldDuration = 0
        end
    end
    instantInteractConnection = RunService.Heartbeat:Connect(function()
        if not instantInteractActive then return end
        for prompt, _ in pairs(originalHoldDurations) do
            if prompt and prompt.Parent and prompt.HoldDuration ~= 0 then
                prompt.HoldDuration = 0
            end
        end
    end)
    instantInteractTrackConnection = game.DescendantAdded:Connect(onDescendantAdded)
end

local function disableInstantInteract()
    if not instantInteractActive then return end
    instantInteractActive = false
    if instantInteractConnection then instantInteractConnection:Disconnect() end
    if instantInteractTrackConnection then instantInteractTrackConnection:Disconnect() end
    restoreAllPrompts()
    originalHoldDurations = {}
end

local flyActive = false
local flyBodyGyro = nil
local flyBodyVelocity = nil
local flySpeed = 50
local flyCurrentSpeed = 0
local flyMaxSpeed = 50
local flyConnection = nil

local function startFly()
    local char = LocalPlayer.Character
    if not char then return end
    local torso = char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
    local hum = char:FindFirstChild("Humanoid")
    if not torso or not hum then return end
    if flyBodyGyro then flyBodyGyro:Destroy() end
    if flyBodyVelocity then flyBodyVelocity:Destroy() end
    if flyConnection then flyConnection:Disconnect() end
    flyBodyGyro = Instance.new("BodyGyro", torso)
    flyBodyGyro.P = 9e4
    flyBodyGyro.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    flyBodyGyro.CFrame = torso.CFrame
    flyBodyVelocity = Instance.new("BodyVelocity", torso)
    flyBodyVelocity.Velocity = Vector3.new(0, 0.1, 0)
    flyBodyVelocity.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    hum.PlatformStand = true
    local animate = char:FindFirstChild("Animate")
    if animate then animate.Disabled = true end
    flyCurrentSpeed = 0
    flyMaxSpeed = flySpeed
    flyConnection = RunService.RenderStepped:Connect(function()
        if not flyActive then return end
        local currentChar = LocalPlayer.Character
        if not currentChar then return end
        local currentTorso = currentChar:FindFirstChild("UpperTorso") or currentChar:FindFirstChild("Torso")
        local currentHum = currentChar:FindFirstChild("Humanoid")
        if not currentTorso or not currentHum then return end
        local moveDir = currentHum.MoveDirection
        local cam = Workspace.CurrentCamera
        local forward = cam.CFrame.LookVector
        local right = cam.CFrame.RightVector
        local moveVector = Vector3.new(moveDir.X, 0, moveDir.Z)
        local moveDirectionWorld = (forward * -moveVector.Z) + (right * moveVector.X)
        if moveVector.Magnitude > 0 then
            moveDirectionWorld = moveDirectionWorld.Unit
        else
            moveDirectionWorld = Vector3.new(0, 0, 0)
        end
        if moveVector.Magnitude > 0 then
            flyCurrentSpeed = math.min(flyCurrentSpeed + 0.5 + (flyCurrentSpeed / flyMaxSpeed), flyMaxSpeed)
        else
            flyCurrentSpeed = math.max(flyCurrentSpeed - 1, 0)
        end
        flyBodyVelocity.Velocity = moveDirectionWorld * flyCurrentSpeed
        if moveVector.Magnitude > 0 then
            flyBodyGyro.CFrame = CFrame.new(currentTorso.Position, currentTorso.Position + moveDirectionWorld)
        else
            flyBodyGyro.CFrame = cam.CFrame
        end
    end)
end

local function stopFly()
    flyActive = false
    if flyBodyGyro then flyBodyGyro:Destroy() end
    if flyBodyVelocity then flyBodyVelocity:Destroy() end
    if flyConnection then flyConnection:Disconnect() end
    flyBodyGyro = nil
    flyBodyVelocity = nil
    flyConnection = nil
    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChild("Humanoid")
        if hum then hum.PlatformStand = false end
        local animate = char:FindFirstChild("Animate")
        if animate then animate.Disabled = false end
    end
    flyCurrentSpeed = 0
end

local fullBrightState = false

local function setFullBright(enabled)
    if enabled then
        savedLighting = {
            Ambient = Lighting.Ambient,
            OutdoorAmbient = Lighting.OutdoorAmbient,
            Brightness = Lighting.Brightness,
            FogEnd = Lighting.FogEnd,

        Lighting.Brightness = 2
        Lighting.FogEnd = 1000000
        Lighting.GlobalShadows = false
        Lighting.ClockTime = 12
        for _, v in pairs(Lighting:GetChildren()) do
            if v:IsA("BloomEffect") or v:IsA("Atmosphere") or v:IsA("Clouds") or v:IsA("Fog") then
                v.Enabled = false
            end
        end
    else
        if savedLighting then
            Lighting.Ambient = savedLighting.Ambient
            Lighting.OutdoorAmbient = savedLighting.OutdoorAmbient
            Lighting.Brightness = savedLighting.Brightness
            Lighting.FogEnd = savedLighting.FogEnd
            Lighting.GlobalShadows = savedLighting.GlobalShadows

        end
        for _, v in pairs(Lighting:GetChildren()) do
            if v:IsA("BloomEffect") or v:IsA("Atmosphere") or v:IsA("Clouds") then
                v.Enabled = true
            end
        end
    end
end

local function startCaseFarm()
    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local rootPart = character:WaitForChild("HumanoidRootPart")
    local miscFolder = Workspace:FindFirstChild("Misc")
    if not miscFolder then
        StarterGui:SetCore("SendNotification", { Title = "Farm Status", Text = "Folder 'Misc' not found!", Duration = 3 })
        return
    end
    local targetNames = {
        "AILACase", "MurchCase", "EventCase", "EVENTCase", "EVENTCASE",
        "BLACKLIGHTCase", "BlacklightCase", "Blacklightcase",
        "blacklightcase", "blacklightCase", "blacklight case", "BlackLightCase", "BlackLightcase"
    }
    local foundCases = {}
    for _, obj in ipairs(miscFolder:GetChildren()) do
        for _, name in ipairs(targetNames) do
            if obj.Name == name then
                table.insert(foundCases, obj)
                break
            end
        end
    end
    if #foundCases == 0 then
        StarterGui:SetCore("SendNotification", { Title = "Farm Status", Text = "0 cases found", Duration = 5 })
        return
    end
    for _, case in ipairs(foundCases) do
        if case:IsA("BasePart") or case:FindFirstChildWhichIsA("BasePart") then
            local targetPos = case:IsA("BasePart") and case.CFrame or case:FindFirstChildWhichIsA("BasePart").CFrame
            rootPart.CFrame = targetPos * CFrame.new(0, 3, 0)
            task.wait(2.2)
        end
    end
    StarterGui:SetCore("SendNotification", { Title = "Farm Status", Text = "Finished farming all found cases!", Duration = 3 })
end

--  ANTI-CHEAT BYPASS
-- ============================================================
local bypassEnabled = false
local realWalkSpeed = 16
local fakeWalkSpeed = 16

local function enableBypass()
    if bypassEnabled then return end
    bypassEnabled = true
    local oldNamecall
    oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
        local method = getnamecallmethod()
        if not checkcaller() and method == "Kick" then
            return nil
        end
        return oldNamecall(self, ...)
    end)
    local mt = getrawmetatable(game)
    local oldNewIndex = mt.__newindex
    setreadonly(mt, false)
    mt.__newindex = newcclosure(function(t, k, v)
        if not checkcaller() and t:IsA("Humanoid") then
            if k == "WalkSpeed" then
                realWalkSpeed = v
                return oldNewIndex(t, k, fakeWalkSpeed)
            elseif k == "JumpPower" then
                return nil
            end
        end
        return oldNewIndex(t, k, v)
    end)
    local oldIndex = mt.__index
    mt.__index = newcclosure(function(t, k)
        if not checkcaller() and t:IsA("Humanoid") and k == "WalkSpeed" then
            return fakeWalkSpeed
        end
        return oldIndex(t, k)
    end)
    setreadonly(mt, true)
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("LocalScript") then
            local name = v.Name:lower()
            if name:find("anticheat") or name:find("adoni") or name:find("checker") or name:find("speed") then
                v.Disabled = true
            end
        end
    end
end

local function disableBypass()
    bypassEnabled = false
end

-- Collision Fix (отключает коллизию с другими игроками)
local collisionFixActive = false
local collisionFixConnection = nil

Tabs.Movement:AddToggle("Collision Fix", { Title = "Collision Fix", Default = false, Callback = function(state)
    collisionFixActive = state
    if collisionFixConnection then
        collisionFixConnection:Disconnect()
        collisionFixConnection = nil
    end
    if state then
        collisionFixConnection = RunService.Stepped:Connect(function()
            if not collisionFixActive then return end
            for _, player in pairs(Players:GetPlayers()) do
                if player ~= LocalPlayer and player.Character then
                    for _, part in pairs(player.Character:GetDescendants()) do
                        if part:IsA("BasePart") then
                            part.CanCollide = false
                        end
                    end
                end
            end
        end)
    else
        -- Восстанавливаем коллизию при выключении
        for _, player in pairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and player.Character then
                for _, part in pairs(player.Character:GetDescendants()) do
                    if part:IsA("BasePart") then
                        part.CanCollide = true
                    end
                end
            end
        end
    end
end })
